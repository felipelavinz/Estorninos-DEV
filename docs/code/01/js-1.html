function confirmar(){
	if ( confirm('Desea continuar') ) {
		alert('Aceptó los términos y condiciones');
	} else {
		alert('No aceptó :-(');
	}
}

Retornar valores distintos de acuerdo al input del usuario

function retornar( valor ) {
	switch(valor){
		case 0:
			return 'no hay comentarios';
			break;
		case 1:
			return 'un comentario';
			break;
		default:
			return valor.toString() +' comentarios';
			break;
	}
}

Validar inputs del usuario

function dividir( dividendo, divisor ) {
	if ( typeof(dividendo) !== 'number' || typeof(divisor) !== 'number' ) {
		console.log('Ambos parámetros deben ser números');
		return null;
	}
	return dividendo / divisor;
}

function sumar2( sumando1, sumando2 ) {
	/**
	 * También puede ser:
	 *  return parseInt(sumando1) + parseInt(sumando2);
	 *  */
	return ( sumando1 * 1 ) + ( sumando2 * 1 );
}
