Crear elementos

function crear_elementos( cantidad ) {
	var sandbox = document.getElementById('child_container');
	for ( i = 0; i < cantidad; i++ ) {
		var new_element = document.createElement('div');
		new_element.id = 'elemento-'+ i;
		new_element.className = 'elemento-creado';
		new_element.style.position = 'absolute';
		new_element.style.top = Math.random() * 100 +'%';
		new_element.style.left = Math.random() * 100 +'%';
		new_element.style.backgroundColor = rand_color();
		sandbox.appendChild( new_element );
	}
}

function rand_color(){
	var rr = parseInt( Math.random() * 255 ),
		gg = parseInt( Math.random() * 255 ),
		bb = parseInt( Math.random() * 255 );
	return 'rgb('+ rr +', '+ gg +', '+ bb +')';
}
